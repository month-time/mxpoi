一个简单的live2d插件

2018-04-18

根据官方SDK整合制作的live2d插件，具体效果 可以直接 看我blog 官方原本的demo写的太乱了，这里做了一下整理 使用只需要在 页面内写入 <canvas id="glcanvas" width="200" height="200"></canvas> <script>var live2DHelper = new Live2DHelper({canvas: 'glcanvas'});</script> 即可载入 live2d了，支持点击鼠标右键切换模型 模型数据在LAppDefine.js文件中 可以根据我提供的例子直接添加。 下载地址： live2d-helper.js 下面 是一个带有语言框的example：

document.body.appendChild(live2d_dom);
var path = “/live2d/assets/models/haru/haru.model.json”;

live2DHelper.loadModel(path, function() {
live2DHelper.startTurnHead();
document.body.addEventListener(“mousemove”, followPointer, false);
document.body.onclick = function() {
live2DHelper.setRandomExpression();
if (document.querySelector(“.message”).classList[1]) {
document.querySelector(“.message”).classList.remove(“hoves”);
} else {
document.querySelector(“.message”).classList.add(“hoves”);
}
}
});
function followPointer(e) {
live2DHelper.followPointer(e);
}
